CREATE TABLE TC_PDF(
	ID_PDF INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	NOMBRE_PDF NOT NULL VARCHAR(100),
	URL_PDF NOT NULL VARCHAR(100),
	FECHA_CREACION DATETIME,
	ACTIVO BIT DEFAULT 1,
);


CREATE TABLE TC_ROL(
	ID_ROL INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	NOMBRE_ROL NOT NULL VARCHAR(40),
	ACTIVO BIT DEFAULT 1,
);

CREATE TABLE TC_MODULO (
	ID_MODULO INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	NOMBRE_MODULO NOT NULL VARCHAR(40),
	ACTIVO BIT DEFAULT 1,
);

CREATE TABLE TC_MODULO_ROL (
	ID_MODULO INT NOT NULL REFERENCES TC_MODULO,
	ID_ROL INT NOT NULL REFERENCES TC_ROL,
	ACTIVO BIT DEFAULT 1,
	PRIMARY KEY(ID_MODULO, ID_ROL)
);


CREATE TABLE TC_USUARIO(
	ID_USUARIO INT IDENTITY(1,1) NOT NULL, 
	USUARIO_NICK VARCHAR(40) NOT NULL UNIQUE,
	NOMBRE VARCHAR(40) NOT NULL,
	APELLIDO_PATERNO VARCHAR(40) NOT NULL,
	CORREO VARCHAR(40) NOT NULL UNIQUE,
	PASSWORD VARCHAR(100) NOT NULL,
	ID_ROL INT NOT NULL,
	ACTIVO BIT DEFAULT 1,
	FOREIGN KEY(ID_ROL) REFERENCES TC_ROL(ID_ROL)
);
